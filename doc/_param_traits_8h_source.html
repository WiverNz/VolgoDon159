45  static T HardMax() { return std::numeric_limits<T>::max(); }
46  static T SoftMin() { return std::numeric_limits<T>::is_signed ? max(T(-1000), HardMin()) : HardMin(); }

48  static T MinStep() { return std::numeric_limits<T>::is_integer ? T(1) : std::numeric_limits<T>::epsilon(); }
